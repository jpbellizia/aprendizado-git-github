echo - Apresenta um texto no console
cd - navegar para outra pasta
cd .. - volta para pasta anterior
cd ../.. - volta duas pastas anteriores
less [Nome do arquivo] - Detalha o que está escrito no conteúdo do arquivo
ls - Mostra todos arquivos presentes na pasta
rm - remove um arquivo
rm -rf - remove uma pasta
history - mostra o histórico de comandos já utilizados no terminal
